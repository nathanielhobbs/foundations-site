4.1 Write a function countGreater that counts how many numbers in a list are greater than a given value k.

def countGreater(L, k):
    count = 0
    for x in L:
        if x > k:
            count += 1
    return count

4.2 Write a function sumList that returns the sum of all numbers in a list.

def sumList(L):
    total = 0
    for x in L:
        total += x
    return total

4.3 Write a function reverseList that returns a new list with the elements reversed.

def reverseList(L):
    out = []
    for i in range(len(L) - 1, -1, -1):
        out.append(L[i])
    return out

def countOdd(L):
    count = 0
    for x in L:
        if x % 2 == 1:
            count += 1
    return count

4.2 Write a function maxDiff that returns the difference between the largest and smallest value in a list.

def maxDiff(L):
    biggest = L[0]
    smallest = L[0]

    for x in L:
        if x > biggest:
            biggest = x
        if x < smallest:
            smallest = x

    return biggest - smallest

def elements_at_odd_indices(L):
